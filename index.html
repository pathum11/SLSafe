import os, time, json, threading, requests import pandas as pd import numpy as np from binance.client import Client from flask import Flask, request # --- 1. CONFIGURATIONS --- API_KEY = os.getenv("BINANCE_API_KEY") API_SECRET = os.getenv("BINANCE_API_SECRET") TELEGRAM_TOKEN = os.getenv("TELEGRAM_TOKEN") TELEGRAM_CHAT_ID = os.getenv("TELEGRAM_CHAT_ID") PORT = int(os.environ.get("PORT", 8080)) client = Client(API_KEY, API_SECRET) app = Flask(__name__) DB_FILE = "trade_state.json" # Settings LB_BARS = 5 RB_BARS = 5 DEVIATION_PCT = 1.5 ATR_LEN = 11 ATR_MULT_TP = 4.0 ATR_MULT_SL = 4.0 MAX_POSITIONS = 15 def load_data(): if os.path.exists(DB_FILE): try: with open(DB_FILE, 'r') as f: return json.load(f) except: pass return { 'active_positions': {}, 'symbol_accumulated_loss': {}, 'signal_count': 0, 'bot_memory': {}, 'is_paused': False, # Telegram notifications pause 'is_scanning': True # Market scanning control (නව එක) } state = load_data() # කලින් state එකේ is_scanning නැතිනම් එය එකතු කිරීම if 'is_scanning' not in state: state['is_scanning'] = True def sync_save(): try: with open(DB_FILE, 'w') as f: json.dump(state, f) except: pass def execute_telegram_send(msg): if state.get('is_paused', False): return # පෝස් කර ඇත්නම් මැසේජ් යවන්නේ නැත url = f"https://api.telegram.org/bot{TELEGRAM_TOKEN}/sendMessage" try: requests.post(url, data={"chat_id": TELEGRAM_CHAT_ID, "text": msg, "parse_mode": "HTML"}, timeout=15) except: pass # --- 2. INDICATOR LOGIC (HH/LL/HL/LH) --- def get_pivots_python(df, lb, rb): highs = df['high'].values lows = df['low'].values p_highs = [np.nan] * len(df) p_lows = [np.nan] * len(df) if len(df) < (lb + rb + 2): return p_highs, p_lows for i in range(lb, len(df) - rb): if all(highs[i] >= highs[i-j] for j in range(1, lb+1)) and all(highs[i] > highs[i+j] for j in range(1, rb+1)): p_highs[i] = highs[i] if all(lows[i] <= lows[i-j] for j in range(1, lb+1)) and all(lows[i] < lows[i+j] for j in range(1, rb+1)): p_lows[i] = lows[i] return p_highs, p_lows def check_signal(symbol, df): try: if symbol not in state['bot_memory']: state['bot_memory'][symbol] = {'lookLH': False, 'lookHL': False} p_highs, p_lows = get_pivots_python(df, LB_BARS, RB_BARS) valid_ph = [x for x in p_highs if not np.isnan(x)] valid_pl = [x for x in p_lows if not np.isnan(x)] if len(valid_ph) < 2 or len(valid_pl) < 2: return None last_ph, prev_ph = valid_ph[-1], valid_ph[-2] last_pl, prev_pl = valid_pl[-1], valid_pl[-2] dev = df['close'].iloc[-1] * (DEVIATION_PCT / 100) if (last_ph > prev_ph) and (last_ph - prev_ph > dev): state['bot_memory'][symbol]['lookLH'] = True if (last_pl < prev_pl) and (prev_pl - last_pl > dev): state['bot_memory'][symbol]['lookHL'] = True if (last_ph < prev_ph) and state['bot_memory'][symbol]['lookLH']: state['bot_memory'][symbol]['lookLH'] = False return "SELL" if (last_pl > prev_pl) and state['bot_memory'][symbol]['lookHL']: state['bot_memory'][symbol]['lookHL'] = False return "BUY" except: pass return None # --- 3. CORE ENGINE --- def open_trade(s, side, df): if not state.get('is_scanning', True): return # ස්කෑන් නවතා ඇත්නම් ට්‍රේඩ් ගන්නේ නැත curr_p = df['close'].iloc[-1] df['tr'] = np.maximum(df['high'] - df['low'], np.maximum(abs(df['high'] - df['close'].shift(1)), abs(df['low'] - df['close'].shift(1)))) atr = df['tr'].rolling(ATR_LEN).mean().iloc[-1] loss_rec = state['symbol_accumulated_loss'].get(s, 0.0) t_dist = (atr * ATR_MULT_TP) + loss_rec s_dist = (atr * ATR_MULT_SL) tp = curr_p + t_dist if side == "BUY" else curr_p - t_dist sl = curr_p - s_dist if side == "BUY" else curr_p + s_dist state['active_positions'][s] = {'side': side, 'tp': tp, 'sl': sl, 'entry': curr_p} state['signal_count'] += 1 execute_telegram_send(f"🚀 SIGNAL #{state['signal_count']}\n🪙 {s} | {side}\n🎯 TP: {round(tp, 6)}\n🛑 SL: {round(sl, 6)}") sync_save() def scan_markets(): while True: try: # පවතින ට්‍රේඩ් පාලනය සැමවිටම සිදු වේ (Close කිරීමට) for s in list(state['active_positions'].keys()): try: p = float(client.futures_symbol_ticker(symbol=s)['price']) pos = state['active_positions'][s] if (pos['side']=="BUY" and p >= pos['tp']) or (pos['side']=="SELL" and p <= pos['tp']): state['symbol_accumulated_loss'][s] = 0.0 execute_telegram_send(f"✅ PROFIT {s}") del state['active_positions'][s] sync_save() elif (pos['side']=="BUY" and p <= pos['sl']) or (pos['side']=="SELL" and p >= pos['sl']): state['symbol_accumulated_loss'][s] = state['symbol_accumulated_loss'].get(s, 0.0) + abs(pos['entry'] - pos['sl']) execute_telegram_send(f"❌ LOSS {s}") del state['active_positions'][s] sync_save() except: continue # ස්කෑන් කිරීම ක්‍රියාත්මක නම් පමණක් අලුත් ට්‍රේඩ් පරීක්ෂා කරයි if state.get('is_scanning', True): tickers = client.futures_ticker() symbols = [t['symbol'] for t in tickers if t['symbol'].endswith("USDT") and float(t['lastPrice']) < 1.0] for s in symbols: if not state.get('is_scanning', True): break # අතරමැදදී නැවැත්වුවහොත් if len(state['active_positions']) >= MAX_POSITIONS: break if s in state['active_positions']: continue k = client.futures_klines(symbol=s, interval='5m', limit=500) df = pd.DataFrame(k, columns=['t','o','high','low','close','v','ct','qv','nt','tb','tq','i']) df[['high','low','close']] = df[['high','low','close']].astype(float) side = check_signal(s, df) if side: open_trade(s, side, df) time.sleep(0.5) time.sleep(10) except: time.sleep(10) # --- 4. TELEGRAM WEBHOOK (COMMANDS) --- @app.route('/webhook', methods=['POST']) def webhook(): update = request.get_json() if "message" in update and "text" in update["message"]: text = update["message"]["text"] # 1. පණිවිඩ පාලනය (පැරණි එක) if text == "/pause": state['is_paused'] = True sync_save() execute_telegram_send("🔇 Notifications Paused.") elif text == "/resume": state['is_paused'] = False sync_save() execute_telegram_send("🔊 Notifications Resumed.") # 2. ස්කෑන් කිරීම පාලනය (අලුත් එක) elif text == "/scan_stop": state['is_scanning'] = False sync_save() execute_telegram_send("🛑 SCAN STOPPED: No new signals or recovery trades will be taken.") elif text == "/scan_start": state['is_scanning'] = True sync_save() execute_telegram_send("▶️ SCAN STARTED: Searching for new signals and recovery trades.") return "OK", 200 @app.route('/') def home(): status = "RUNNING" if state.get('is_scanning', True) else "STOPPED" return f"Bot Status: {status}", 200 if __name__ == "__main__": threading.Thread(target=scan_markets, daemon=True).start() app.run(host='0.0.0.0', port=PORT)
